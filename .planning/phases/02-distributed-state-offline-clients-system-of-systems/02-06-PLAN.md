    ```swift
    /// Canonical runtime state terminology for Vel operator surfaces.
    /// These constants must be used across all Apple targets to ensure
    /// consistent language with web dashboard, CLI, and documentation.
    public enum VelStatusTerminology {
        public static let fresh = "fresh"
        public static let stale = "stale"
        public static let degraded = "degraded"
        public static let offline = "offline"
        public static let recovering = "recovering"
        public static let denied = "denied"

        /// Human-readable descriptions for each status (for display in UI)
        public static func description(for status: String) -> String {
            switch status {
            case fresh: return "Connected and up-to-date"
            case stale: return "Connected but data may be outdated"
            case degraded: return "Connection issue — some features may be limited"
            case offline: return "Offline — using cached data"
            case recovering: return "Reconnecting..."
            case denied: return "Access denied by capability policy"
            default: return "Status unknown"
            }
        }

        /// Recovery hint text for non-fresh states
        public static func recoveryHint(for status: String) -> String? {
            switch status {
            case stale: return "Pull to refresh or check your connection."
            case degraded: return "Check network or run `vel doctor` for diagnostics."
            case offline: return "Reconnect to sync new changes."
            case denied: return "Contact your system administrator or review capability settings."
            default: return nil
            }
        }
    }
    ```

    ```markdown
    # Vel Troubleshooting Guide

    ## Sync Status: degraded

    **Symptoms**: Dashboard shows "degraded" status badge. Some data may be missing or stale.

    **Recovery steps**:
    1. Run `vel doctor` — check the diagnostic output for specific failure reasons
    2. Run `vel config show` — verify endpoint URLs are correct
    3. Restart veld: `make dev-api`
    4. Check veld logs for ERROR-level entries

    ## Sync Status: offline

    **Symptoms**: No real-time updates. Data shows last cached state.

    **Recovery steps**:
    1. Verify veld is running: `curl http://localhost:4130/api/health`
    2. Check `vel.toml` bind_addr matches the URL your clients are using
    3. Restart veld

    ## Pairing / Node Link Failed

    **Symptoms**: `vel node link` fails or pairing token expires before use.

    **Recovery steps**:
    1. Pairing tokens expire after 5 minutes. Run `vel node link` again.
    2. Ensure both devices are on the same network (LAN URL) or Tailscale
    3. Run `vel node list` to verify the new node appears after pairing

    ## Capability Denied

    **Symptoms**: Agent or integration action fails with "denied" status.

    **Recovery steps**:
    1. Run `vel connect list` to inspect agent capability allowlists
    2. Launch the agent with the required capability in its allowlist
    3. Capability scopes: read:context, write:captures, read:commitments, execute:commands
    ```
